—Publish .net app—
dotnet publish -o /Users/alex/Desktop/RateMyTalk
(must be in root of app directory)


—Use SCP to copy published folder—
scp -r /Users/alex/Desktop/RateMyTalk/* root@<IP_ADDRESS>:/var/www


—Create the service file—
sudo nano /etc/systemd/system/kestrel-ratemytalk.service

[Unit]
Description=RateMyTalk

[Service]
WorkingDirectory=/var/www
ExecStart=/usr/bin/dotnet /var/www/RateMyTalk.dll
Restart=always
RestartSec=10  # Restart service after 10 seconds if dotnet service crashes
SyslogIdentifier=dotnet-RateMyTalk
User=www-data
Environment=ASPNETCORE_ENVIRONMENT=Production 

[Install]
WantedBy=multi-user.target


—Save the file and enable the service—
systemctl enable kestrel-ratemytalk.service


—Start the service and verify that it is running—
systemctl start kestrel-ratemytalk.service
systemctl status kestrel-ratemytalk.service
